  6 /**

  8  * This is a function for the load pokemon data from the API.
  9  *  
 10  * @param {number} quantity - This is the number of pokemon that are loaded.
 11  * @param {number} state - This is a status state in a number.
 12  * 
 13  */
 14 async function loadPokemonData(quantity, state) {
 15     waitAnimationShow(true);
 16     document.getElementById("mainDiv").innerHTML = "";
 17     for (let index = 1; index < quantity + 1; index++) {
 18         await writePokemonData(index, state);    
 19     }
 20     setTimeout(() => waitAnimationShow(false), 1200);
 21 }

 56 async function writePokemonData(int, state) {
 57     let responseData = await loadPokemonDataFromApi(BASE_URL + int.toString());
 58     let typesOfPokemon = searchTypesOfPokemon(responseData);
 59     if (isNumberLowerAsInt(!state, 1)) {
 60         allPokemonNames.push(responseData.name);
 61     }
 62     document.getElementById("mainDiv").innerHTML += renderPokemon(responseData, typesOfPokemon, int);
 63     
 64 }
 65 /**
 66  * 
 67  * This is a function for search the types from all pokemon.
 68  *  
 69  * @param {object} responseData - This is the fetch data from API.
 70  *
 71  */
 72 function searchTypesOfPokemon(responseData) {
 73     let typesOfPokemon = [];
 74     for (let index = 0; index < responseData.types.length; index++) {
 75         let typeOfPokemon = responseData.types[index];
 76         typesOfPokemon.push(typeOfPokemon.type.name);
 77     }
 78     return typesOfPokemon;
 79 }

141     let filterWord = document.getElementById("pokemonSearch").value;
142     if (filterWord.length < 3) {
143         notFoundFilter("remove");
144         document.getElementById("footer").classList.remove("filterFooter");
145         await loadPokemonData(loadedPokemonInt, 0);
146     } else {
147         document.getElementById("mainDiv").innerHTML = "";
148         showFilterArray();
149     }

151 /**
152  * 
153  * This is a function for the show data of the filter array.
154  *  
155  */
156 async function showFilterArray() {
157     let filterArray = await filterThePokemon(document.getElementById("pokemonSearch").value);    
158     if (isNumberLowerAsInt(filterArray.length, 1)) {
159         notFoundFilter("add");
160         document.getElementById("footer").classList.add("filterFooter");
161     } else if (isNumberLowerAsInt(filterArray.length, 10)) {
162         notFoundFilter("remove");
163         document.getElementById("footer").classList.add("filterFooter");
164         addContentInFilterArray(filterArray);
165     } else {
166         notFoundFilter("remove");
167         document.getElementById("footer").classList.add("filterFooter");
168     }
169 }
